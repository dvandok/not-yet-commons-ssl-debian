032    package org.apache.commons.ssl.rmi;
033    
034    import java.io.Serializable;
035    import java.rmi.Remote;
036    import java.rmi.RemoteException;
037    import java.rmi.server.UnicastRemoteObject;
038    import java.util.Date;
039    
040    /**
041     * @author Credit Union Central of British Columbia
042     * @author <a href="http://www.cucbc.com/">www.cucbc.com</a>
043     * @author <a href="mailto:juliusdavies@cucbc.com">juliusdavies@cucbc.com</a>
044     * @since 22-Feb-2007
045     */
046    public class DateRMI extends UnicastRemoteObject
047        implements Remote, Serializable, RemoteDate {
048        private Date d;
049    
050        public DateRMI() throws RemoteException {
051            super();
052            this.d = new Date();
053        }
054    
055        public Date getDate() throws RemoteException {
056            return d;
057        }
058    
059        public boolean equals(Object o) {
060            RemoteDate rd = (RemoteDate) o;
061            try {
062                return d.equals(rd.getDate());
063            }
064            catch (RemoteException re) {
065                return false;
066            }
067        }
068    
069    }
